Integration

HTML (auto-init)

<!-- Include the script -->
<script src="zam-pay-button.js"></script>

<!-- Add a container with data attributes -->
<div class="zam-pay-button"
     data-type="pay"
     data-style="dark"
     data-size="default"></div>

JavaScript API

// Render programmatically with click handler
ZamPayButton.render(document.getElementById('my-btn'), {
  type: 'pay',
  style: 'dark',
  size: 'default',
  onClick: function() {
    ZamCheckout.open({ amount: '15.500' });
  }
});

CSS Custom Properties

/* Override button dimensions from your stylesheet */
.zam-pay-btn {
  --zam-btn-radius: 12px;
  --zam-btn-min-width: 200px;
  --zam-btn-height: 48px;
}
